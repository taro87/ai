CREATE TABLE MAJOR(
  mCODE NUMBER(2) PRIMARY KEY,
  mNAME VARCHAR2(15),
mOFFICE VARCHAR2(10)
);
CREATE TABLE STUDENT(
  sNO NUMBER(3) PRIMARY KEY,
  sNAME VARCHAR2(12),
 sSCORE NUMBER(3) CHECK(sSCORE<=100),
  mCODE NUMBER(2)
  );
